import { describe, it, expect } from 'vitest';
import { render, screen } from '@testing-library/react';
import { BrowserRouter } from 'react-router-dom';
import ProjectCard from '../ProjectCard';

const mockProject = {
  id: 1,
  name: 'Test Project',
  description: 'Test Description',
  status: 'draft' as const,
  created_at: '2024-01-15T10:00:00Z',
  updated_at: '2024-01-15T10:00:00Z',
};

describe('ProjectCard', () => {
  it('renders project name and description', () => {
    render(
      <BrowserRouter>
        <ProjectCard project={mockProject} />
      </BrowserRouter>
    );
    
    expect(screen.getByText('Test Project')).toBeInTheDocument();
    expect(screen.getByText('Test Description')).toBeInTheDocument();
  });

  it('displays status badge', () => {
    render(
      <BrowserRouter>
        <ProjectCard project={mockProject} />
      </BrowserRouter>
    );
    
    expect(screen.getByText('draft')).toBeInTheDocument();
  });

  it('shows different status colors', () => {
    const completedProject = { ...mockProject, status: 'completed' as const };
    render(
      <BrowserRouter>
        <ProjectCard project={completedProject} />
      </BrowserRouter>
    );
    
    const statusBadge = screen.getByText('completed');
    expect(statusBadge).toHaveClass('status-completed');
  });
});

import { describe, it, expect, beforeEach, vi } from 'vitest';
import { useProjectStore } from '../projectStore';
import { projectsApi } from '../../api/projects';

// Mock API
vi.mock('../../api/projects', () => ({
  projectsApi: {
    getAll: vi.fn(),
    getById: vi.fn(),
    create: vi.fn(),
    update: vi.fn(),
    delete: vi.fn(),
  },
}));

describe('ProjectStore', () => {
  beforeEach(() => {
    useProjectStore.setState({
      projects: [],
      selectedProject: null,
      loading: false,
      error: null,
    });
    vi.clearAllMocks();
  });

  it('fetches projects successfully', async () => {
    const mockProjects = [
      { id: 1, name: 'Project 1', description: 'Desc 1', status: 'draft', created_at: '', updated_at: '' },
    ];
    vi.mocked(projectsApi.getAll).mockResolvedValue(mockProjects);

    await useProjectStore.getState().fetchProjects();

    expect(useProjectStore.getState().projects).toEqual(mockProjects);
    expect(useProjectStore.getState().loading).toBe(false);
    expect(useProjectStore.getState().error).toBe(null);
  });

  it('handles fetch error', async () => {
    vi.mocked(projectsApi.getAll).mockRejectedValue(new Error('Network error'));

    await useProjectStore.getState().fetchProjects();

    expect(useProjectStore.getState().error).toBe('Network error');
    expect(useProjectStore.getState().loading).toBe(false);
  });

  it('creates a new project', async () => {
    const newProject = { id: 2, name: 'New Project', description: '', status: 'draft', created_at: '', updated_at: '' };
    vi.mocked(projectsApi.create).mockResolvedValue(newProject);

    await useProjectStore.getState().createProject({ name: 'New Project' });

    expect(useProjectStore.getState().projects).toContainEqual(newProject);
  });

  it('deletes a project', async () => {
    useProjectStore.setState({
      projects: [
        { id: 1, name: 'Project 1', description: '', status: 'draft', created_at: '', updated_at: '' },
      ],
    });

    await useProjectStore.getState().deleteProject(1);

    expect(useProjectStore.getState().projects).toHaveLength(0);
  });
});
