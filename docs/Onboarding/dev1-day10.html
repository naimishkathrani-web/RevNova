import { SalesforceService } from '../salesforce.service';

describe('SalesforceService', () => {
  it('should connect with valid credentials', async () => {
    const service = new SalesforceService();
    const result = await service.connect('user', 'pass', 'token');
    expect(result.success).toBe(true);
  });

  it('should describe object metadata', async () => {
    const service = new SalesforceService();
    const metadata = await service.describeObject('Account');
    expect(metadata.fields).toBeDefined();
  });
});

üß™ Step 3: Write Integration Tests for Endpoints

File: backend/src/routes/__tests__/analyze.test.ts

import request from 'supertest';
import app from '../../app';

describe('POST /projects/:id/analyze', () => {
  it('should queue schema analysis job', async () => {
    const res = await request(app)
      .post('/api/v1/projects/1/analyze')
      .send({ objects: ['Account'] });
    expect(res.status).toBe(202);
    expect(res.body.jobId).toBeDefined();
  });
});
