Create Docker Compose configuration for local development with backend, frontend, PostgreSQL, and Redis services.

üê≥ Step 1: Create docker-compose.yml

File: docker-compose.yml

version: '3.8'

services:
  postgres:
    image: postgres:16
    environment:
      POSTGRES_USER: revnova
      POSTGRES_PASSWORD: dev123
      POSTGRES_DB: revnova_dev
    ports: ['5432:5432']
    volumes: ['./data/postgres:/var/lib/postgresql/data']

  redis:
    image: redis:7
    ports: ['6379:6379']

  backend:
    build: ./backend
    ports: ['3000:3000']
    depends_on: [postgres, redis]
    environment:
      DATABASE_URL: postgres://revnova:dev123@postgres:5432/revnova_dev
      REDIS_URL: redis://redis:6379

  frontend:
    build: ./frontend
    ports: ['5173:5173']
