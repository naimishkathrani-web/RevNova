Day 2: Database Tables (STG1)

Developer 1: Backend & Database | Duration: ~6-8 hours

üìã Objective

Create the foundational STG1 database tables for the RevNova migration system. These tables will store project information, Salesforce connections, and source schema data (objects and fields from SFDC).

Tables to Create:

  • projects - Migration project metadata
  • connections - Salesforce connection credentials
  • source_objects - SFDC objects (Product2, PricebookEntry, etc.)
  • source_fields - SFDC fields with full metadata

ÔøΩ Step 1: Create Projects Table Migration

üí¨ GitHub Copilot Prompt:

# Create database migration file for STG1 tables
# Define projects table schema: id, name, description, status, timestamps
# Include proper indexes and constraints

Create a new migration file:

cd backend
npm run migration:create -- create-projects-table

File: backend/migrations/001_create_projects_table.sql

-- Projects table: stores migration project information
CREATE TABLE projects (
  id SERIAL PRIMARY KEY,
  name VARCHAR(255) NOT NULL,
  description TEXT,
  source_type VARCHAR(50) NOT NULL DEFAULT 'salesforce', -- salesforce, netsuite, etc.
  target_type VARCHAR(50) NOT NULL DEFAULT 'salesforce',
  status VARCHAR(50) NOT NULL DEFAULT 'draft', -- draft, analyzing, mapping, ready, executing, completed, failed
  created_by VARCHAR(255),
  created_at TIMESTAMP NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMP NOT NULL DEFAULT NOW()
);

-- Index for faster queries
CREATE INDEX idx_projects_status ON projects(status);
CREATE INDEX idx_projects_created_at ON projects(created_at DESC);

‚úÖ Checkpoint: Projects table migration file created

ÔøΩ Step 2: Create Connections Table Migration

üí¨ GitHub Copilot Prompt:

# Define connections table: id, project_id, name, type, credentials, status
# Add foreign key to projects table
# Include encrypted credentials field

File: backend/migrations/002_create_connections_table.sql

-- Connections table: stores Salesforce connection credentials
CREATE TABLE connections (
  id SERIAL PRIMARY KEY,
  project_id INTEGER NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
  name VARCHAR(255) NOT NULL,
  type VARCHAR(50) NOT NULL DEFAULT 'salesforce', -- salesforce, custom
  instance_url VARCHAR(500) NOT NULL, -- https://yourorg.my.salesforce.com
  access_token TEXT, -- encrypted token
  refresh_token TEXT, -- encrypted token
  client_id VARCHAR(255),
  client_secret TEXT, -- encrypted
  username VARCHAR(255),
  is_sandbox BOOLEAN DEFAULT true,
  status VARCHAR(50) NOT NULL DEFAULT 'pending', -- pending, connected, failed
  last_tested_at TIMESTAMP,
  created_at TIMESTAMP NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMP NOT NULL DEFAULT NOW()
);

-- Index for project lookups
CREATE INDEX idx_connections_project_id ON connections(project_id);
CREATE INDEX idx_connections_status ON connections(status);

üì¶ Step 3: Create Source Objects Table Migration

File: backend/migrations/003_create_source_objects_table.sql

-- Source Objects table: stores Salesforce object metadata
CREATE TABLE source_objects (
  id SERIAL PRIMARY KEY,
  connection_id INTEGER NOT NULL REFERENCES connections(id) ON DELETE CASCADE,
  name VARCHAR(255) NOT NULL, -- e.g., Product2, PricebookEntry
  label VARCHAR(255) NOT NULL, -- e.g., Product, Pricebook Entry
  api_name VARCHAR(255) NOT NULL, -- fully qualified API name
  is_custom BOOLEAN DEFAULT false,
  is_queryable BOOLEAN DEFAULT true,
  is_createable BOOLEAN DEFAULT true,
  is_updateable BOOLEAN DEFAULT true,
  is_deletable BOOLEAN DEFAULT true,
  record_count INTEGER DEFAULT 0,
  key_prefix VARCHAR(10), -- e.g., 01t for Product2
  description TEXT,
  created_at TIMESTAMP NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMP NOT NULL DEFAULT NOW(),
  UNIQUE(connection_id, api_name)
);

-- Indexes for performance
CREATE INDEX idx_source_objects_connection_id ON source_objects(connection_id);
CREATE INDEX idx_source_objects_name ON source_objects(name);
CREATE INDEX idx_source_objects_is_custom ON source_objects(is_custom);

File: backend/migrations/004_create_source_fields_table.sql

-- Source Fields table: stores Salesforce field metadata
CREATE TABLE source_fields (
  id SERIAL PRIMARY KEY,
  source_object_id INTEGER NOT NULL REFERENCES source_objects(id) ON DELETE CASCADE,
  name VARCHAR(255) NOT NULL, -- e.g., Name, ProductCode
  label VARCHAR(255) NOT NULL, -- e.g., Product Name
  api_name VARCHAR(255) NOT NULL,
  type VARCHAR(100) NOT NULL, -- string, number, boolean, date, datetime, picklist, reference, etc.
  length INTEGER,
  precision_value INTEGER, -- renamed from 'precision' to avoid keyword
  scale_value INTEGER, -- renamed from 'scale'
  is_required BOOLEAN DEFAULT false,
  is_unique BOOLEAN DEFAULT false,
  is_external_id BOOLEAN DEFAULT false,
  is_custom BOOLEAN DEFAULT false,
  default_value TEXT,
  picklist_values JSONB, -- array of {label, value}
  reference_to VARCHAR(255), -- referenced object name for lookup fields
  relationship_name VARCHAR(255),
  is_calculated BOOLEAN DEFAULT false,
  formula TEXT,
  help_text TEXT,
  created_at TIMESTAMP NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMP NOT NULL DEFAULT NOW(),
  UNIQUE(source_object_id, api_name)
);

-- Indexes for queries
CREATE INDEX idx_source_fields_object_id ON source_fields(source_object_id);
CREATE INDEX idx_source_fields_name ON source_fields(name);
CREATE INDEX idx_source_fields_type ON source_fields(type);
CREATE INDEX idx_source_fields_is_custom ON source_fields(is_custom);
