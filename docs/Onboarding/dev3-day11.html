üß™ Step 2: Write Connection Tests

File: backend/src/routes/__tests__/connection.test.ts

import request from 'supertest';
import app from '../../app';

describe('Connection API', () => {
  test('POST /connections creates connection', async () => {
    const res = await request(app)
      .post('/api/v1/connections')
      .send({ provider: 'salesforce', username: 'test@example.com' });
    expect(res.status).toBe(201);
  });

  test('POST /connections/test validates credentials', async () => {
    const res = await request(app).post('/api/v1/connections/test').send({ username: 'test', password: 'pass' });
    expect(res.body).toHaveProperty('success');
  });
});
