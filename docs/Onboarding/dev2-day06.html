Day 6: Salesforce Connections UI

Developer 2: Frontend | Duration: ~6-8 hours

üìã Objective

Build the Salesforce connection management interface for RevNova. Users will be able to create new Salesforce connections, view existing connections, and test connectivity before saving.

UI Components to Build:

  • Connection form with authentication fields (username, password, security token, environment)
  • Connection list view showing all saved connections with status indicators
  • Test connection button with real-time validation feedback
  • Error handling and success notifications

Design Reference: Follow RevNova's Salesforce-inspired design system from docs/styles.css and form patterns from docs/login.html.

File: frontend/src/components/connections/ConnectionForm.tsx

import React, { useState } from 'react';
import axios from 'axios';
import './ConnectionForm.css';

interface ConnectionFormData {
  name: string;
  username: string;
  password: string;
  securityToken: string;
  environment: 'production' | 'sandbox';
}

interface ConnectionFormProps {
  onSuccess: () => void;
}

export const ConnectionForm: React.FC<ConnectionFormProps> = ({ onSuccess }) => {
  const [formData, setFormData] = useState<ConnectionFormData>({
    name: '',
    username: '',
    password: '',
    securityToken: '',
    environment: 'production'
  });
  const [testing, setTesting] = useState(false);
  const [saving, setSaving] = useState(false);
  const [testResult, setTestResult] = useState<{ success: boolean; message: string } | null>(null);
  const [error, setError] = useState<string | null>(null);

  const handleChange = (e: React.ChangeEvent<HTMLInputElement | HTMLSelectElement>) => {
    const { name, value } = e.target;
    setFormData(prev => ({ ...prev, [name]: value }));
    setTestResult(null); // Clear test result when form changes
  };

  const handleTestConnection = async () => {
    setTesting(true);
    setTestResult(null);
    setError(null);

    try {
      const response = await axios.post('/api/v1/connections/test', {
        username: formData.username,
        password: formData.password,
        securityToken: formData.securityToken,
        environment: formData.environment
      });

      setTestResult({
        success: true,
        message: `Connected successfully! Organization: ${response.data.organizationId}`
      });
    } catch (err: any) {
      setTestResult({
        success: false,
        message: err.response?.data?.error || 'Connection test failed'
      });
    } finally {
      setTesting(false);
    }
  };

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setSaving(true);
    setError(null);

    try {
      await axios.post('/api/v1/connections', formData);
      onSuccess();
      // Reset form
      setFormData({
        name: '',
        username: '',
        password: '',
        securityToken: '',
        environment: 'production'
      });
      setTestResult(null);
    } catch (err: any) {
      setError(err.response?.data?.error || 'Failed to save connection');
    } finally {
      setSaving(false);
    }
  };

  return (
    <div className="connection-form-container">
      <h2>New Salesforce Connection</h2>
      
      <form onSubmit={handleSubmit} className="connection-form">
        <div className="form-group">
          <label htmlFor="name">Connection Name</label>
          <input
            type="text"
            id="name"
            name="name"
            value={formData.name}
            onChange={handleChange}
            placeholder="e.g., Production Org"
            required
          />
        </div>

        <div className="form-group">
          <label htmlFor="environment">Environment</label>
          <select
            id="environment"
            name="environment"
            value={formData.environment}
            onChange={handleChange}
            required
          >
            <option value="production">Production</option>
            <option value="sandbox">Sandbox</option>
          </select>
        </div>

        <div className="form-group">
          <label htmlFor="username">Salesforce Username</label>
          <div className="input-with-icon">
            <i className="fas fa-user"></i>
            <input
              type="text"
              id="username"
              name="username"
              value={formData.username}
              onChange={handleChange}
              placeholder="user@example.com"
              required
            />
          </div>
        </div>

        <div className="form-group">
          <label htmlFor="password">Password</label>
          <div className="input-with-icon">
            <i className="fas fa-lock"></i>
            <input
              type="password"
              id="password"
              name="password"
              value={formData.password}
              onChange={handleChange}
              placeholder="Enter password"
              required
            />
          </div>
        </div>

        <div className="form-group">
          <label htmlFor="securityToken">Security Token</label>
          <div className="input-with-icon">
            <i className="fas fa-key"></i>
            <input
              type="text"
              id="securityToken"
              name="securityToken"
              value={formData.securityToken}
              onChange={handleChange}
              placeholder="Enter security token"
              required
            />
          </div>
          <small className="form-hint">
            Don't have your security token? 
            <a href="https://help.salesforce.com/s/articleView?id=sf.user_security_token.htm" target="_blank" rel="noopener noreferrer">
              Learn how to reset it
            </a>
          </small>
        </div>

        {testResult && (
          <div className={`test-result ${testResult.success ? 'success' : 'error'}`}>
            <i className={`fas ${testResult.success ? 'fa-check-circle' : 'fa-exclamation-circle'}`}></i>
            {testResult.message}
          </div>
        )}

        {error && (
          <div className="error-message">
            <i className="fas fa-exclamation-triangle"></i>
            {error}
          </div>
        )}

        <div className="form-actions">
          <button
            type="button"
            className="btn btn-secondary"
            onClick={handleTestConnection}
            disabled={testing || !formData.username || !formData.password}
          >
            {testing ? (
              <>
                <i className="fas fa-spinner fa-spin"></i> Testing...
              </>
            ) : (
              <>
                <i className="fas fa-plug"></i> Test Connection
              </>
            )}
          </button>

          <button
            type="submit"
            className="btn btn-primary"
            disabled={saving || !testResult?.success}
          >
            {saving ? (
              <>
                <i className="fas fa-spinner fa-spin"></i> Saving...
              </>
            ) : (
              <>
                <i className="fas fa-save"></i> Save Connection
              </>
            )}
          </button>
        </div>
      </form>
    </div>
  );
};

File: frontend/src/components/connections/ConnectionList.tsx

import React, { useEffect, useState } from 'react';
import axios from 'axios';
import './ConnectionList.css';

interface Connection {
  id: string;
  name: string;
  username: string;
  environment: 'production' | 'sandbox';
  createdAt: string;
}

export const ConnectionList: React.FC = () => {
  const [connections, setConnections] = useState<Connection[]>([]);
  const [loading, setLoading] = useState(true);
  const [testingId, setTestingId] = useState<string | null>(null);
  const [testResults, setTestResults] = useState<Record<string, { success: boolean; message: string }>>({});

  useEffect(() => {
    fetchConnections();
  }, []);

  const fetchConnections = async () => {
    try {
      const response = await axios.get('/api/v1/connections');
      setConnections(response.data.connections);
    } catch (error) {
      console.error('Failed to fetch connections:', error);
    } finally {
      setLoading(false);
    }
  };

  const handleTestConnection = async (connectionId: string) => {
    setTestingId(connectionId);
    
    try {
      const response = await axios.post(`/api/v1/connections/${connectionId}/test`);
      setTestResults(prev => ({
        ...prev,
        [connectionId]: {
          success: true,
          message: 'Connection successful'
        }
      }));
    } catch (error: any) {
      setTestResults(prev => ({
        ...prev,
        [connectionId]: {
          success: false,
          message: error.response?.data?.error || 'Test failed'
        }
      }));
    } finally {
      setTestingId(null);
    }
  };

  if (loading) {
    return (
      <div className="connection-list-container">
        <div className="loading-spinner">
          <i className="fas fa-spinner fa-spin"></i> Loading connections...
        </div>
      </div>
    );
  }

  if (connections.length === 0) {
    return (
      <div className="connection-list-container">
        <div className="empty-state">
          <i className="fas fa-plug fa-3x"></i>
          <h3>No connections yet</h3>
          <p>Create your first Salesforce connection to get started</p>
        </div>
      </div>
    );
  }

  return (
    <div className="connection-list-container">
      <h2>Saved Connections</h2>
      
      <div className="connection-grid">
        {connections.map(connection => (
          <div key={connection.id} className="connection-card">
            <div className="connection-header">
              <div className="connection-icon">
                <i className="fas fa-cloud"></i>
              </div>
              <div className="connection-info">
                <h3>{connection.name}</h3>
                <span className={`env-badge ${connection.environment}`}>
                  {connection.environment}
                </span>
              </div>
            </div>

            <div className="connection-details">
              <div className="detail-row">
                <span className="detail-label">Username:</span>
                <span className="detail-value">{connection.username}</span>
              </div>
              <div className="detail-row">
                <span className="detail-label">Created:</span>
                <span className="detail-value">
                  {new Date(connection.createdAt).toLocaleDateString()}
                </span>
              </div>
            </div>

            {testResults[connection.id] && (
              <div className={`connection-test-result ${testResults[connection.id].success ? 'success' : 'error'}`}>
                <i className={`fas ${testResults[connection.id].success ? 'fa-check-circle' : 'fa-times-circle'}`}></i>
                {testResults[connection.id].message}
              </div>
            )}

            <div className="connection-actions">
              <button
                className="btn btn-test"
                onClick={() => handleTestConnection(connection.id)}
                disabled={testingId === connection.id}
              >
                {testingId === connection.id ? (
                  <>
                    <i className="fas fa-spinner fa-spin"></i> Testing...
                  </>
                ) : (
                  <>
                    <i className="fas fa-plug"></i> Test
                  </>
                )}
              </button>
            </div>
          </div>
        ))}
      </div>
    </div>
  );
};
