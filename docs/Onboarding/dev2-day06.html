Build the Connection Wizard form (Step 1 of 7-page flow) with source/target connection inputs, referencing mockup/migration-connect.html.

üé® Step 1: Create ConnectionForm Component

File: frontend/src/components/ConnectionForm.tsx

import React, { useState } from 'react';

export const ConnectionForm = () => {
  const [connectionName, setConnectionName] = useState('');
  const [username, setUsername] = useState('');
  const [password, setPassword] = useState('');
  const [securityToken, setSecurityToken] = useState('');

  return (
    <div className="max-w-2xl mx-auto p-6">
      <h2 className="text-2xl font-bold mb-4">Connect to Salesforce</h2>
      <input
        type="text"
        placeholder="Connection Name"
        value={connectionName}
        onChange={(e) => setConnectionName(e.target.value)}
        className="w-full p-2 border rounded mb-3"
      />
      <input type="text" placeholder="Username" value={username} onChange={(e) => setUsername(e.target.value)} className="w-full p-2 border rounded mb-3" />
      <input type="password" placeholder="Password" value={password} onChange={(e) => setPassword(e.target.value)} className="w-full p-2 border rounded mb-3" />
      <input type="text" placeholder="Security Token" value={securityToken} onChange={(e) => setSecurityToken(e.target.value)} className="w-full p-2 border rounded mb-3" />
    </div>
  );
};
