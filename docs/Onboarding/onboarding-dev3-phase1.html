Day 2 (Tuesday): Lint & Test Workflow

Duration: 8 hours
Tasks:
  1. Create .github/workflows/ci.yml workflow
  2. Configure lint job (ESLint for backend + frontend)
  3. Configure test job (Jest for backend, Vitest for frontend)
  4. Run on pull requests and main branch pushes
  5. Add status badge to README.md
💡 Copilot Prompt: "Create GitHub Actions workflow for Node.js 20 monorepo that runs ESLint and Jest tests in parallel, with caching for node_modules"
CI Workflow Example:
# .github/workflows/ci.yml
name: CI

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main, develop]

jobs:
  lint:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'
      
      - name: Install dependencies
        run: npm ci
      
      - name: Lint backend
        run: cd backend && npm run lint
      
      - name: Lint frontend
        run: cd frontend && npm run lint

  test-backend:
    runs-on: ubuntu-latest
    services:
      postgres:
        image: postgres:15
        env:
          POSTGRES_PASSWORD: testpass
        options: >-
          --health-cmd pg_isready
          --health-interval 10s
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'
      
      - name: Install dependencies
        run: cd backend && npm ci
      
      - name: Run tests
        run: cd backend && npm test
        env:
          DATABASE_URL: postgresql://postgres:testpass@localhost:5432/test

  test-frontend:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'
      
      - name: Install dependencies
        run: cd frontend && npm ci
      
      - name: Run tests
        run: cd frontend && npm test
✅ Testing: Push to feature branch and verify workflow runs successfully on GitHub Actions tab
Deliverable:
  • ✓ CI workflow running on every PR

Week 2: Docker Compose & Local Development (Nov 11-15, 2025)

Goal: Set up local development environment with Docker Compose (PostgreSQL, Redis, backend, frontend)

Day 6 (Monday): Docker Compose Configuration

Duration: 8 hours
Tasks:
  1. Create docker-compose.yml in project root
  2. Define services: PostgreSQL 15, Redis 7, backend, frontend
  3. Configure volumes for data persistence
  4. Set up networking between services
  5. Create .env.example file
💡 Copilot Prompt: "Create Docker Compose file for Node.js backend, React frontend, PostgreSQL 15, and Redis 7 with health checks and proper networking"
Docker Compose Example:
version: '3.9'

services:
  postgres:
    image: postgres:15
    environment:
      POSTGRES_USER: revnova
      POSTGRES_PASSWORD: devpass
      POSTGRES_DB: revnova_dev
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U revnova"]
      interval: 10s
      timeout: 5s
      retries: 5

  redis:
    image: redis:7-alpine
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s

  backend:
    build:
      context: ./backend
      dockerfile: Dockerfile.dev
    ports:
      - "3000:3000"
    environment:
      DATABASE_URL: postgresql://revnova:devpass@postgres:5432/revnova_dev
      REDIS_URL: redis://redis:6379
      NODE_ENV: development
    volumes:
      - ./backend:/app
      - /app/node_modules
    depends_on:
      postgres:
        condition: service_healthy
      redis:
        condition: service_healthy

  frontend:
    build:
      context: ./frontend
      dockerfile: Dockerfile.dev
    ports:
      - "5173:5173"
    volumes:
      - ./frontend:/app
      - /app/node_modules
    depends_on:
      - backend

volumes:
  postgres_data:
  redis_data:
Deliverable:
  • ✓ Docker Compose file complete
  • ✓ All services start with docker compose up

Integration Testing Best Practices

API Integration Tests (Supertest)

// tests/integration/connection.test.ts
import request from 'supertest';
import { app } from '../../src/app';
import { setupTestDb, teardownTestDb } from '../helpers/db';

describe('POST /api/v1/connections/test', () => {
  beforeAll(async () => {
    await setupTestDb();
  });

  afterAll(async () => {
    await teardownTestDb();
  });

  it('should validate SFDC connection', async () => {
    const response = await request(app)
      .post('/api/v1/connections/test')
      .send({
        instanceUrl: 'https://test.salesforce.com',
        accessToken: 'mock_token'
      })
      .expect(200);

    expect(response.body.success).toBe(true);
    expect(response.body.orgId).toBeDefined();
  });

  it('should return 400 for invalid credentials', async () => {
    const response = await request(app)
      .post('/api/v1/connections/test')
      .send({
        instanceUrl: 'invalid-url'
      })
      .expect(400);

    expect(response.body.error).toBeDefined();
  });
});

E2E Tests (Playwright)

// tests/e2e/connection-wizard.spec.ts
import { test, expect } from '@playwright/test';

test.describe('Connection Wizard', () => {
  test('should complete connection flow', async ({ page }) => {
    await page.goto('http://localhost:5173/wizard/connection');
    
    // Select environment
    await page.selectOption('#environment', 'sandbox');
    
    // Enter credentials
    await page.fill('#instanceUrl', 'https://test.salesforce.com');
    await page.fill('#username', 'test@example.com');
    await page.fill('#password', 'testpass');
    
    // Submit
    await page.click('button[type="submit"]');
    
    // Verify success
    await expect(page.locator('.success-message')).toBeVisible();
    await expect(page).toHaveURL('/wizard/analyze');
  });
});
