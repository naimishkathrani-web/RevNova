üí° Step 2: Create AI Service

import OpenAI from 'openai';

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

export async function suggestMappings(sourceFields: any[], targetFields: any[]) {
  const prompt = `Given these source fields from SFDC CPQ: ${JSON.stringify(sourceFields)}
And target fields from Revenue Cloud: ${JSON.stringify(targetFields)}
Suggest the best field mappings with confidence scores (0.0-1.0). Return JSON array.`;

  const response = await openai.chat.completions.create({
    model: 'gpt-4',
    messages: [{ role: 'user', content: prompt }],
    response_format: { type: 'json_object' }
  });

  return JSON.parse(response.choices[0].message.content);
}

üåê Step 3: Create AI Suggestions Endpoint

router.post('/projects/:id/mappings/suggest', async (req, res) => {
  const sourceFields = await getSourceFields(req.params.id);
  const targetFields = await getTargetFields(req.params.id);

  const suggestions = await suggestMappings(sourceFields, targetFields);
  res.json({ suggestions });
});
