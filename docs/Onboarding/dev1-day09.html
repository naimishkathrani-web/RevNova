üîó Step 1: Create Relationships Table

File: backend/src/database/migrations/009_create_relationships.sql

CREATE TABLE object_relationships (
  id SERIAL PRIMARY KEY,
  project_id INTEGER REFERENCES projects(id),
  source_object VARCHAR(255),
  source_field VARCHAR(255),
  target_object VARCHAR(255),
  relationship_type VARCHAR(50), -- 'lookup', 'master-detail'
  cascade_delete BOOLEAN DEFAULT false,
  discovered_at TIMESTAMP DEFAULT NOW()
);

ü§ñ Step 2: Detect Relationships from Metadata

File: backend/src/services/relationship.service.ts

export class RelationshipService {
  async detectRelationships(projectId: number, metadata: any) {
    const relationships = [];

    for (const field of metadata.fields) {
      if (field.type === 'reference' && field.referenceTo) {
        for (const target of field.referenceTo) {
          relationships.push({
            projectId,
            sourceObject: metadata.name,
            sourceField: field.name,
            targetObject: target,
            relationshipType: field.relationshipType || 'lookup',
            cascadeDelete: field.cascadeDelete || false
          });
        }
      }
    }

    return relationships;
  }
}

üìä Step 3: Create Relationship Graph Endpoint

File: backend/src/routes/analyze.routes.ts

router.get('/projects/:id/relationships', async (req, res) => {
  const { id } = req.params;

  const query = `
    SELECT source_object, source_field, target_object, relationship_type
    FROM object_relationships
    WHERE project_id = $1
  `;

  const result = await db.query(query, [id]);
  res.json({ relationships: result.rows });
});
