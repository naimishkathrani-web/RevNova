    try {
      const { instanceUrl, accessToken } = req.body;
      const sf = new SalesforceService(instanceUrl, accessToken);
      const orgInfo = await sf.testConnection();
      res.json({ success: true, org: orgInfo });

      res.status(400).json({ success: false, error: error.message });
    }
  }
}

Service Structure:
// src/services/metadataExtractionService.ts
export class MetadataExtractionService {
  async extractObjectMetadata(objectName: string) {
    const metadata = await this.sf.conn.describe(objectName);
    return {
      name: metadata.name,
      label: metadata.label,
      fields: metadata.fields.map(f => ({
        name: f.name,
        type: f.type,
        label: f.label,
        required: !f.nillable,
        custom: f.custom,
        referenceTo: f.referenceTo
      }))
    };
  }
}

VS Code + Copilot Best Practices

Daily Copilot Workflow

  1. Start with prompts: Open Copilot Chat, describe the feature you're building
  2. Inline suggestions: Start typing function signature, let Copilot suggest implementation
  3. Test generation: Ask "Generate unit tests for this function" in Chat
  4. Refactoring: Select code, ask "Refactor this to improve error handling"
  5. Documentation: Ask "Add JSDoc comments explaining this service class"

Git Commit Best Practices

  • Use conventional commits: feat:, fix:, chore:, test:
  • Reference issue numbers: Closes #123
  • Keep commits atomic (one logical change per commit)
  • Write descriptive messages explaining WHY, not just WHAT
