Step 1: Create ExecutionProgress Component

File: frontend/src/components/ExecutionProgress.tsx

import { useEffect, useState } from 'react';
import { useNavigate } from 'react-router-dom';
import { api } from '../utils/api';

interface ExecutionStatus {
  status: 'EXECUTING' | 'PAUSED' | 'COMPLETED' | 'FAILED';
  progress: number;
  phase: 'EXTRACTING' | 'TRANSFORMING' | 'LOADING';
  recordsProcessed: number;
  totalRecords: number;
  errorMessage?: string;
}

export function ExecutionProgress({ migrationId }: { migrationId: string }) {
  const [status, setStatus] = useState<ExecutionStatus | null>(null);
  const [polling, setPolling] = useState(true);
  const navigate = useNavigate();

  useEffect(() => {
    if (!polling) return;

    const pollStatus = async () => {
      try {
        const response = await api.get(`/migrations/${migrationId}/status`);
        setStatus(response.data);

        if (response.data.status === 'COMPLETED') {
          setPolling(false);
          setTimeout(() => navigate(`/migrations/${migrationId}/report`), 2000);
        } else if (response.data.status === 'FAILED') {
          setPolling(false);
        }
      } catch (error) {
        console.error('Failed to fetch status:', error);
      }
    };

    const interval = setInterval(pollStatus, 2000);
    pollStatus(); // Initial call

    return () => clearInterval(interval);
  }, [migrationId, polling, navigate]);

  if (!status) return <div>Loading...</div>;

  const phaseIcon = {
    EXTRACTING: 'üîç',
    TRANSFORMING: '‚öôÔ∏è',
    LOADING: 'üì§'
  }[status.phase];

  const phaseColor = {
    EXTRACTING: 'bg-blue-500',
    TRANSFORMING: 'bg-purple-500',
    LOADING: 'bg-green-500'
  }[status.phase];

  return (
    <div className="max-w-4xl mx-auto p-6">
      <h1 className="text-3xl font-bold mb-6">Migration in Progress</h1>

      <div className="bg-white rounded-lg shadow-lg p-8">
        {/* Progress Bar */}
        <div className="mb-8">
          <div className="flex justify-between mb-2">
            <span className="text-lg font-semibold">
              {phaseIcon} {status.phase}
            </span>
            <span className="text-lg font-semibold">{status.progress}%</span>
          </div>
          
          <div className="w-full bg-gray-200 rounded-full h-4">
            <div
              className={`${phaseColor} h-4 rounded-full transition-all duration-500 ease-out`}
              style={{ width: `${status.progress}%` }}
            />
          </div>
        </div>

        {/* Records Count */}
        <div className="mb-6">
          <p className="text-gray-700 text-lg">
            Processing <strong>{status.recordsProcessed.toLocaleString()}</strong> of{' '}
            <strong>{status.totalRecords.toLocaleString()}</strong> records
          </p>
        </div>

        {/* Status Messages */}
        {status.status === 'COMPLETED' && (
          <div className="bg-green-100 border border-green-400 text-green-700 px-4 py-3 rounded">
            ‚úÖ Migration completed successfully!
          </div>
        )}

        {status.status === 'FAILED' && (
          <div className="bg-red-100 border border-red-400 text-red-700 px-4 py-3 rounded">
            ‚ùå Migration failed: {status.errorMessage}
          </div>
        )}

        {/* Controls */}
        {status.status === 'EXECUTING' && (
          <div className="flex gap-4">
            <button
              onClick={() => handleCancel()}
              className="px-6 py-2 bg-red-600 text-white rounded hover:bg-red-700"
            >
              Cancel Migration
            </button>
          </div>
        )}
      </div>
    </div>
  );
}

Step 2: Add Cancel Confirmation Modal

File: frontend/src/components/CancelConfirmModal.tsx

export function CancelConfirmModal({ 
  isOpen, 
  onClose, 
  onConfirm 
}: { 
  isOpen: boolean;
  onClose: () => void;
  onConfirm: () => void;
}) {
  if (!isOpen) return null;

  return (
    <div className="fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center z-50">
      <div className="bg-white rounded-lg p-6 max-w-md">
        <h3 className="text-xl font-bold mb-4">Cancel Migration?</h3>
        <p className="text-gray-700 mb-6">
          Are you sure you want to cancel this migration? This action cannot be undone.
          Any records already migrated will remain in the target system.
        </p>
        <div className="flex gap-4 justify-end">
          <button
            onClick={onClose}
            className="px-4 py-2 border border-gray-300 rounded hover:bg-gray-100"
          >
            No, Continue
          </button>
          <button
            onClick={onConfirm}
            className="px-4 py-2 bg-red-600 text-white rounded hover:bg-red-700"
          >
            Yes, Cancel Migration
          </button>
        </div>
      </div>
    </div>
  );
}
