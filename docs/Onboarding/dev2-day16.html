üìã Objective

Build the Transformation Preview page to call POST /api/transform/preview and display before/after comparison of transformed data with color-coded field changes.

üé® Step 1: Create TransformationPreview Component

Create frontend/src/pages/TransformationPreview.tsx:

import { useState } from 'react';
import { useNavigate } from 'react-router-dom';

interface TransformResult {
  source: Record<string, any>;
  transformed: Record<string, any>;
}

interface PreviewResponse {
  results: TransformResult[];
  errors: any[];
  errorCount: number;
}

export default function TransformationPreview() {
  const navigate = useNavigate();
  const [preview, setPreview] = useState<PreviewResponse | null>(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const runPreview = async () => {
    setLoading(true);
    setError(null);

    try {
      const mappingId = sessionStorage.getItem('mappingId');
      const response = await fetch('/api/transform/preview', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          transforms: JSON.parse(sessionStorage.getItem('transforms') || '[]'),
          sampleData: JSON.parse(sessionStorage.getItem('sampleData') || '[]'),
          connectionId: sessionStorage.getItem('connectionId')
        })
      });

      if (!response.ok) throw new Error('Preview failed');

      const data = await response.json();
      setPreview(data);
    } catch (err: any) {
      setError(err.message);
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="min-h-screen bg-gray-50 p-8">
      <h1 className="text-3xl font-bold mb-6">Transformation Preview</h1>

      <button
        onClick={runPreview}
        disabled={loading}
        className="px-6 py-3 bg-blue-600 text-white rounded-lg hover:bg-blue-700 disabled:opacity-50"
      >
        {loading ? 'Running Preview...' : 'Run Transformation Preview'}
      </button>

      {error && (
        <div className="mt-4 p-4 bg-red-100 text-red-700 rounded">{error}</div>
      )}

      {preview && (
        <div className="mt-6">
          <h2 className="text-xl font-semibold mb-4">
            Preview Results ({preview.results.length} rows, {preview.errorCount} errors)
          </h2>
          {preview.results.map((result, idx) => (
            <PreviewRow key={idx} result={result} />
          ))}
        </div>
      )}
    </div>
  );
}

üîÑ Step 2: Create PreviewRow Component with Before/After Comparison

Create frontend/src/components/PreviewRow.tsx:

interface PreviewRowProps {
  result: {
    source: Record<string, any>;
    transformed: Record<string, any>;
  };
}

export default function PreviewRow({ result }: PreviewRowProps) {
  return (
    <div className="bg-white border rounded-lg p-4 mb-4">
      <div className="grid grid-cols-2 gap-4">
        <div>
          <h3 className="font-semibold mb-2 text-gray-700">Source Data</h3>
          <div className="space-y-1">
            {Object.entries(result.source).map(([key, value]) => (
              <div key={key} className="text-sm">
                <span className="font-medium">{key}:</span> {JSON.stringify(value)}
              </div>
            ))}
          </div>
        </div>

        <div>
          <h3 className="font-semibold mb-2 text-green-700">Transformed Data</h3>
          <div className="space-y-1">
            {Object.entries(result.transformed).map(([key, value]) => {
              const changed = result.source[key] !== value;
              return (
                <div
                  key={key}
                  className={`text-sm ${changed ? 'bg-green-50 p-1 rounded' : ''}`}
                >
                  <span className="font-medium">{key}:</span> {JSON.stringify(value)}
                  {changed && <span className="ml-2 text-green-600">‚úì Changed</span>}
                </div>
              );
            })}
          </div>
        </div>
      </div>
    </div>
  );
}
