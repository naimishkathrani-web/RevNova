üîç Step 2: Add Validation Rules

const TYPE_COMPATIBILITY = {
  string: ['string', 'text', 'textarea', 'url', 'email'],
  number: ['double', 'int', 'currency', 'percent'],
  boolean: ['boolean', 'checkbox'],
  date: ['date', 'datetime']
};

export function validateMapping(sourceField: any, targetField: any) {
  const warnings = [];

  if (!isTypeCompatible(sourceField.type, targetField.type)) {
    warnings.push(`Type mismatch: ${sourceField.type} ‚Üí ${targetField.type}`);
  }

  if (targetField.required && !sourceField.value) {
    warnings.push(`Required field ${targetField.name} has no mapping`);
  }

  return warnings;
}
