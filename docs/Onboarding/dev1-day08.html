CREATE TABLE schema_catalog (
  id SERIAL PRIMARY KEY,
  project_id INTEGER REFERENCES projects(id),
  object_name VARCHAR(255),
  field_name VARCHAR(255),
  field_label VARCHAR(255),
  data_type VARCHAR(50),
  is_custom BOOLEAN,
  is_required BOOLEAN,
  picklist_values JSONB,
  indexed_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_catalog_search ON schema_catalog(project_id, field_name, field_label);

üíæ Step 2: Create Catalog Updater Service

File: backend/src/services/catalog.service.ts

import { Pool } from 'pg';

export class CatalogService {
  constructor(private db: Pool) {}

  async indexObject(projectId: number, metadata: any) {
    const values = [];
    for (const field of metadata.fields) {
      values.push([
        projectId,
        metadata.name,
        field.name,
        field.label,
        field.type,
        field.custom,
        field.required,
        JSON.stringify(field.picklistValues || [])
      ]);
    }

    const query = `INSERT INTO schema_catalog       (project_id, object_name, field_name, field_label, data_type, is_custom, is_required, picklist_values)       VALUES ${values.map((_, i) => `($${i*8+1},$${i*8+2},$${i*8+3},$${i*8+4},$${i*8+5},$${i*8+6},$${i*8+7},$${i*8+8})`).join(',')}`;

    await this.db.query(query, values.flat());
  }
}

üîç Step 3: Create Search Endpoint

File: backend/src/routes/analyze.routes.ts

router.get('/projects/:id/catalog/search', async (req, res) => {
  try {
    const { id } = req.params;
    const { q } = req.query;

    const query = `
      SELECT * FROM schema_catalog
      WHERE project_id = $1
      AND (field_name ILIKE $2 OR field_label ILIKE $2)
      LIMIT 20
    `;

    const results = await db.query(query, [id, `%${q}%`]);
    res.json({ results: results.rows });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});
