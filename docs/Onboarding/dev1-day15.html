describe('Mapping API', () => {
  test('POST /mappings saves 70+ mappings', async () => {
    const res = await request(app).post('/api/v1/projects/1/mappings').send({ mappings: testMappings });
    expect(res.body.count).toBe(70);
  });

  test('POST /mappings/suggest returns AI suggestions', async () => {
    const res = await request(app).post('/api/v1/projects/1/mappings/suggest');
    expect(res.body.suggestions).toBeDefined();
    expect(res.body.suggestions[0].confidence).toBeGreaterThan(0);
  });

  test('POST /mappings/validate detects unmapped required fields', async () => {
    const res = await request(app).post('/api/v1/projects/1/mappings/validate');
    expect(res.body.errors.length).toBeGreaterThan(0);
  });
});
