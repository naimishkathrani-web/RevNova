import { useDroppable } from '@dnd-kit/core';

const DroppableTarget = ({ targetField, onDrop }) => {
  const { setNodeRef } = useDroppable({ id: targetField.name });

  const handleDragEnd = (event) => {

      onDrop(active.id, over.id); // sourceField, targetField
    }
  };

  return <div ref={setNodeRef} className="p-2 border-2 border-dashed">Drop here</div>;
};
