üîê Step 1: Create OAuth Button

File: frontend/src/components/OAuthButton.tsx

export const OAuthButton = () => {
  const handleOAuth = () => {
    const authUrl = `/api/v1/auth/salesforce?redirect_uri=${encodeURIComponent(window.location.origin + '/oauth/callback')}`;
    window.open(authUrl, 'oauth', 'width=600,height=700');
  };

  return (
    <button onClick={handleOAuth} className="bg-blue-600 text-white px-4 py-2 rounded">
      Connect with OAuth
    </button>
  );
};

‚Ü©Ô∏è Step 2: Handle OAuth Callback

File: frontend/src/pages/OAuthCallback.tsx

useEffect(() => {
  const code = new URLSearchParams(window.location.search).get('code');
  if (code) {
    fetch('/api/v1/auth/callback', {
      method: 'POST',
      body: JSON.stringify({ code })
    }).then(() => window.close());
  }
}, []);

return (
