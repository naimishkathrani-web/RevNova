import { render, screen, waitFor, fireEvent } from '@testing-library/react';
import { vi } from 'vitest';
import ValidationResults from '../ValidationResults';

describe('ValidationResults', () => {
  beforeEach(() => {
    global.fetch = vi.fn();
    sessionStorage.setItem('mappingId', 'test-mapping-123');
  });

  it('should display summary cards with correct counts', async () => {
    (global.fetch as any).mockResolvedValue({
      json: async () => ({
        valid: false,
        issues: [
          { severity: 'critical', field: 'Price', message: 'Required field missing' },
          { severity: 'warning', field: 'Description', message: 'Exceeds 255 chars' }
        ]
      })
    });

    render(<ValidationResults />);

    await waitFor(() => {
      expect(screen.getByText('1')).toBeInTheDocument(); // Critical count
      expect(screen.getByText('1')).toBeInTheDocument(); // Warning count
    });
  });

  it('should disable Start Migration button when critical errors exist', async () => {
    (global.fetch as any).mockResolvedValue({
      json: async () => ({
        valid: false,
        issues: [{ severity: 'critical', field: 'Price', message: 'Required' }]
      })
    });

    render(<ValidationResults />);

    await waitFor(() => {
      const button = screen.getByText('Start Migration');
      expect(button).toBeDisabled();
    });
  });

  it('should open ErrorDetailModal when clicking View Details', async () => {
    (global.fetch as any).mockResolvedValue({
      json: async () => ({
        valid: false,
        issues: [{ severity: 'error', field: 'Price', message: 'Invalid type' }]
      })
    });

    render(<ValidationResults />);

    await waitFor(() => {
      const button = screen.getByText('View Details');

    });

});

üß™ Step 2: Write ErrorDetailModal Tests

Create frontend/src/components/__tests__/ErrorDetailModal.test.tsx:

import { render, screen, waitFor, fireEvent } from '@testing-library/react';
import { vi } from 'vitest';
import ErrorDetailModal from '../ErrorDetailModal';

describe('ErrorDetailModal', () => {
  it('should fetch and display affected records', async () => {
    global.fetch = vi.fn().mockResolvedValue({
      json: async () => ({
        field: 'Price',
        records: [
          { id: 'REC-001', value: 'invalid', error: 'Not a number' },
          { id: 'REC-002', value: 'abc', error: 'Not a number' }
        ]
      })
    });

    render(<ErrorDetailModal field="Price" onClose={vi.fn()} />);

    await waitFor(() => {
      expect(screen.getByText('Error Details: Price')).toBeInTheDocument();
      expect(screen.getByText('REC-001')).toBeInTheDocument();
      expect(screen.getByText('Not a number')).toBeInTheDocument();
    });
  });

  it('should call onClose when clicking Close button', async () => {
    const onClose = vi.fn();
    global.fetch = vi.fn().mockResolvedValue({
      json: async () => ({ field: 'Price', records: [] })
    });

    render(<ErrorDetailModal field="Price" onClose={onClose} />);

    await waitFor(() => screen.getByText('Close'));
    fireEvent.click(screen.getByText('Close'));

    expect(onClose).toHaveBeenCalled();
  });
});
