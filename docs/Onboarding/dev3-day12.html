üß™ Step 1: Write Mapping API Tests

describe('Mapping API', () => {
  test('POST /mappings saves 70+ mappings', async () => {
    const mappings = Array(70).fill(null).map((_, i) => ({
      sourceObject: 'Product2',
      sourceField: `Field${i}`,
      targetObject: 'Product__c',
      targetField: `Field${i}__c`,
      mappingType: 'direct'
    }));

    const res = await request(app).post('/api/v1/projects/1/mappings').send({ mappings });
    expect(res.body.count).toBe(70);
  });
});

test('POST /mappings/suggest returns AI suggestions', async () => {
