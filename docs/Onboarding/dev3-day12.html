üß™ Step 1: Write Mapping API Tests

File: backend/src/routes/__tests__/mapping.test.ts

describe('Mapping API', () => {
  test('POST /mappings saves 70+ mappings', async () => {
    const mappings = Array(70).fill(null).map((_, i) => ({
      sourceObject: 'Product2',
      sourceField: `Field${i}`,
      targetObject: 'Product__c',
      targetField: `Field${i}__c`,
      mappingType: 'direct'
    }));

    const res = await request(app).post('/api/v1/projects/1/mappings').send({ mappings });
    expect(res.body.count).toBe(70);
  });
});

ü§ñ Step 2: Mock OpenAI for AI Tests

jest.mock('openai');

test('POST /mappings/suggest returns AI suggestions', async () => {
  const mockSuggestions = [{ sourceField: 'ProductCode', targetField: 'ProductCode__c', confidence: 0.95 }];
  OpenAI.prototype.chat.completions.create.mockResolvedValue({
    choices: [{ message: { content: JSON.stringify({ suggestions: mockSuggestions }) } }]
  });

  const res = await request(app).post('/api/v1/projects/1/mappings/suggest');
  expect(res.body.suggestions).toBeDefined();
});
