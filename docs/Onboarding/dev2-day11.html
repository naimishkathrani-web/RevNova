export const MappingTable = () => {
  const [mappings, setMappings] = useState([]);

  useEffect(() => {
    fetch('/api/v1/projects/1/mappings')
      .then(res => res.json())
      .then(data => setMappings(data.mappings));
  }, []);

  return (
    <table className="w-full border">
      <thead>
        <tr>
          <th>Source Field</th>
          <th>Target Field</th>
          <th>Status</th>
        </tr>
      </thead>
      <tbody>
        {mappings.map(m => (
          <tr key={m.id}>
            <td>{m.source_field}</td>
            <td>{m.target_field || 'Unmapped'}</td>
            <td>{m.target_field ? '‚úÖ Mapped' : '‚ö†Ô∏è Unmapped'}</td>
          </tr>
        ))}
      </tbody>
    </table>
  );
};
