üå± Step 1: Expand Seed Script

File: backend/scripts/seed.ts

import { Pool } from 'pg';

const db = new Pool({ connectionString: process.env.DATABASE_URL });

// Clear existing data
await db.query('TRUNCATE projects, connections, schema_catalog CASCADE');

// Seed projects
const project = await db.query(`INSERT INTO projects (name, status) VALUES ('CPQ to Revenue Cloud', 'active') RETURNING id`);
const projectId = project.rows[0].id;

// Seed connections
await db.query(`INSERT INTO connections (project_id, provider, username, status) VALUES ($1, 'salesforce', 'dev@example.com', 'connected')`, [projectId]);

// Seed schema catalog
await db.query(`INSERT INTO schema_catalog (project_id, object_name, field_name, field_label, data_type) VALUES ($1, 'Product2', 'ProductCode', 'Product Code', 'string')`, [projectId]);

console.log('‚úÖ Seed data created');
