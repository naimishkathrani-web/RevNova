Day 7: Salesforce Schema Analysis API

Developer 1: Backend & Database | Duration: ~6-8 hours

üìã Objective

Build the schema analysis API that uses jsforce to fetch Salesforce object metadata. Retrieve all standard and custom objects, their fields, data types, and relationships for migration analysis.

API Endpoints to Build:

  • POST /api/v1/analyze/objects - Fetch all Salesforce objects
  • GET /api/v1/analyze/objects/:objectName - Get detailed object metadata
  • POST /api/v1/analyze/fields/:objectName - Fetch all fields for an object

üì¶ Step 1: Create Schema Analysis Service

Create a service to interact with Salesforce metadata API:

File: backend/src/services/schema-analysis.service.ts

import jsforce from 'jsforce';
import { connectToSalesforce } from '../utils/salesforce';

export interface SalesforceObject {
  name: string;
  label: string;
  custom: boolean;
  queryable: boolean;
  createable: boolean;
  updateable: boolean;
  deletable: boolean;
  recordCount?: number;
}

export interface SalesforceField {
  name: string;
  label: string;
  type: string;
  length: number | null;
  precision: number | null;
  scale: number | null;
  required: boolean;
  unique: boolean;
  externalId: boolean;
  defaultValue: any;
  picklistValues: Array<{ label: string; value: string; active: boolean }>;
  referenceTo: string[];
  relationshipName: string | null;
  calculated: boolean;
  custom: boolean;
}

export class SchemaAnalysisService {
  /**
   * Fetch all Salesforce objects from an org
   */
  async fetchAllObjects(connectionId: number): Promise<SalesforceObject[]> {
    const conn = await this.getConnection(connectionId);
    
    // Describe global to get all objects
    const describe = await conn.describeGlobal();
    
    const objects: SalesforceObject[] = describe.sobjects
      .filter(obj => obj.queryable) // Only include queryable objects
      .map(obj => ({
        name: obj.name,
        label: obj.label,
        custom: obj.custom,
        queryable: obj.queryable,
        createable: obj.createable,
        updateable: obj.updateable,
        deletable: obj.deletable
      }));

    return objects;
  }

  /**
   * Get detailed metadata for a specific object
   */
  async describeObject(
    connectionId: number,
    objectName: string
  ): Promise<any> {
    const conn = await this.getConnection(connectionId);
    const metadata = await conn.describe(objectName);

    return {
      name: metadata.name,
      label: metadata.label,
      custom: metadata.custom,
      queryable: metadata.queryable,
      createable: metadata.createable,
      updateable: metadata.updateable,
      deletable: metadata.deletable,
      recordTypeInfos: metadata.recordTypeInfos,
      childRelationships: metadata.childRelationships.map(rel => ({
        childSObject: rel.childSObject,
        field: rel.field,
        relationshipName: rel.relationshipName
      })),
      fieldCount: metadata.fields.length
    };
  }

  /**
   * Fetch all fields for a specific object
   */
  async fetchObjectFields(
    connectionId: number,
    objectName: string
  ): Promise<SalesforceField[]> {
    const conn = await this.getConnection(connectionId);
    const metadata = await conn.describe(objectName);

    const fields: SalesforceField[] = metadata.fields.map(field => ({
      name: field.name,
      label: field.label,
      type: field.type,
      length: field.length || null,
      precision: field.precision || null,
      scale: field.scale || null,
      required: !field.nillable && !field.defaultedOnCreate,
      unique: field.unique || false,
      externalId: field.externalId || false,
      defaultValue: field.defaultValue,
      picklistValues: field.picklistValues || [],
      referenceTo: field.referenceTo || [],
      relationshipName: field.relationshipName || null,
      calculated: field.calculated || false,
      custom: field.custom
    }));

    return fields;
  }

  /**
   * Get row count for an object
   */
  async getObjectRowCount(
    connectionId: number,
    objectName: string
  ): Promise<number> {
    const conn = await this.getConnection(connectionId);
    
    try {
      const result = await conn.query(`SELECT COUNT() FROM ${objectName}`);
      return result.totalSize;
    } catch (error) {
      console.error(`Failed to get row count for ${objectName}:`, error);
      return 0;
    }
  }

  /**
   * Helper to establish Salesforce connection
   */
  private async getConnection(connectionId: number): Promise<jsforce.Connection> {
    // Fetch connection from database
    const pool = require('../config/database').default;
    const result = await pool.query(
      `SELECT username, encrypted_password, password_iv, 
              encrypted_security_token, token_iv, environment
       FROM salesforce_connections 
       WHERE id = $1`,
      [connectionId]
    );

    if (result.rows.length === 0) {
      throw new Error('Connection not found');
    }

    const connData = result.rows[0];
    
    // Decrypt credentials
    const crypto = require('crypto');
    const key = Buffer.from(process.env.ENCRYPTION_KEY!, 'hex');
    
    const decryptPassword = (encrypted: string, iv: string): string => {
      const decipher = crypto.createDecipheriv(
        'aes-256-cbc',
        key,
        Buffer.from(iv, 'hex')
      );
      let decrypted = decipher.update(encrypted, 'hex', 'utf8');
      decrypted += decipher.final('utf8');
      return decrypted;
    };

    const password = decryptPassword(connData.encrypted_password, connData.password_iv);
    const securityToken = decryptPassword(connData.encrypted_security_token, connData.token_iv);

    // Connect to Salesforce
    const { conn } = await connectToSalesforce(
      connData.username,
      password,
      securityToken,
      connData.environment
    );

    return conn;
  }
}

üõ£Ô∏è Step 2: Create Analyze API Routes

Create REST API endpoints for schema analysis:

File: backend/src/routes/analyze.routes.ts

import { Router } from 'express';
import { SchemaAnalysisService } from '../services/schema-analysis.service';

const router = Router();
const schemaService = new SchemaAnalysisService();

/**
 * POST /api/v1/analyze/objects
 * Fetch all Salesforce objects for a connection
 */
router.post('/objects', async (req, res) => {
  try {
    const { connectionId } = req.body;

    if (!connectionId) {
      return res.status(400).json({ error: 'connectionId is required' });
    }

    const objects = await schemaService.fetchAllObjects(connectionId);

    res.json({
      success: true,
      count: objects.length,
      objects: objects
    });
  } catch (error: any) {
    console.error('Failed to fetch objects:', error);
    res.status(500).json({ 
      error: 'Failed to fetch Salesforce objects',
      message: error.message 
    });
  }
});

/**
 * GET /api/v1/analyze/objects/:objectName
 * Get detailed metadata for a specific object
 */
router.get('/objects/:objectName', async (req, res) => {
  try {
    const { objectName } = req.params;
    const { connectionId } = req.query;

    if (!connectionId) {
      return res.status(400).json({ error: 'connectionId query parameter is required' });
    }

    const metadata = await schemaService.describeObject(
      parseInt(connectionId as string),
      objectName
    );

    res.json({
      success: true,
      object: metadata
    });
  } catch (error: any) {
    console.error(`Failed to describe object ${req.params.objectName}:`, error);
    res.status(500).json({ 
      error: 'Failed to fetch object metadata',
      message: error.message 
    });
  }
});

/**
 * POST /api/v1/analyze/fields/:objectName
 * Fetch all fields for a specific object
 */
router.post('/fields/:objectName', async (req, res) => {
  try {
    const { objectName } = req.params;
    const { connectionId } = req.body;

    if (!connectionId) {
      return res.status(400).json({ error: 'connectionId is required' });
    }

    const fields = await schemaService.fetchObjectFields(connectionId, objectName);
    const rowCount = await schemaService.getObjectRowCount(connectionId, objectName);

    res.json({
      success: true,
      objectName: objectName,
      recordCount: rowCount,
      fieldCount: fields.length,
      fields: fields
    });
  } catch (error: any) {
    console.error(`Failed to fetch fields for ${req.params.objectName}:`, error);
    res.status(500).json({ 
      error: 'Failed to fetch field metadata',
      message: error.message 
    });
  }
});

export default router;
