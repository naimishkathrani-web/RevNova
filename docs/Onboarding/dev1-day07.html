üîß Step 2: Create Redis Service

import { createClient } from 'redis';

export class RedisService {
  private client;

  async connect() {
    this.client = createClient({
      url: process.env.REDIS_URL || 'redis://localhost:6379'
    });
    await this.client.connect();
  }

  async set(key: string, value: any, expirySeconds = 3600) {
    await this.client.set(key, JSON.stringify(value), { EX: expirySeconds });
  }

  async get(key: string) {
    const data = await this.client.get(key);
    return data ? JSON.parse(data) : null;
  }
}

üöÄ Step 3: Implement Job Status Endpoint

File: backend/src/routes/analyze.routes.ts

Add this route to check analysis job status:

router.get('/projects/:id/analyze/:jobId', async (req, res) => {
  try {
    const { id, jobId } = req.params;
    
    // Check job status in Redis
    const redisService = new RedisService();
    await redisService.connect();
    const jobStatus = await redisService.get(`job:${jobId}`);

    if (!jobStatus) {
      return res.status(404).json({ error: 'Job not found' });
    }

    res.json(jobStatus);
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});
