‚úÖ Step 1: Create Validation Endpoint

File: backend/src/routes/mapping.routes.ts

router.post('/projects/:id/mappings/validate', async (req, res) => {
  const mappings = await db.query('SELECT * FROM field_mappings WHERE project_id = $1', [req.params.id]);
  const targetFields = await getTargetFields(req.params.id);

  const errors = [];
  const warnings = [];

  // Check required fields
  for (const field of targetFields.filter(f => f.required)) {
    const mapped = mappings.rows.find(m => m.target_field === field.name);
    if (!mapped) {
      errors.push(`Required field ${field.name} is not mapped`);
    }
  }

  // Check type compatibility
  for (const mapping of mappings.rows) {
    const validation = validateMapping(mapping.source_field, mapping.target_field);
    warnings.push(...validation);
  }

  res.json({ valid: errors.length === 0, errors, warnings });
});
