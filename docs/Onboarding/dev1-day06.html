import jsforce from 'jsforce';

export interface SalesforceCredentials {
  instanceUrl: string;
  username: string;
  password: string;
  securityToken?: string;
}

export const connectToSalesforce = async (credentials: SalesforceCredentials) => {
  const conn = new jsforce.Connection({
    loginUrl: credentials.instanceUrl,
  });

  try {
    const passwordWithToken = credentials.securityToken 
      ? credentials.password + credentials.securityToken
      : credentials.password;

    await conn.login(credentials.username, passwordWithToken);
    
    return {
      success: true,
      connection: conn,
      userId: conn.userInfo?.id,
      organizationId: conn.userInfo?.organizationId,
    };
  } catch (error: any) {
    return {
      success: false,
      error: error.message || 'Failed to connect to Salesforce',
    };
  }
};

export const testConnection = async (credentials: SalesforceCredentials): Promise<boolean> => {
  const result = await connectToSalesforce(credentials);
  return result.success;
};

import express from 'express';
import { Pool } from 'pg';
import { connectToSalesforce, testConnection } from '../utils/salesforce';
import crypto from 'crypto';

const router = express.Router();
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

// Encryption key (should be in .env)
const ENCRYPTION_KEY = process.env.ENCRYPTION_KEY || 'default-32-char-encryption-key!';

const encrypt = (text: string): string => {
  const iv = crypto.randomBytes(16);
  const cipher = crypto.createCipheriv('aes-256-cbc', Buffer.from(ENCRYPTION_KEY), iv);
  let encrypted = cipher.update(text, 'utf8', 'hex');
  encrypted += cipher.final('hex');
  return iv.toString('hex') + ':' + encrypted;
};

// POST /api/v1/connections - Create connection
router.post('/', async (req, res) => {
  try {
    const { name, type, instance_url, username, password, security_token } = req.body;
    
    // Validation
    if (!name || !type || !instance_url || !username || !password) {
      return res.status(400).json({ error: 'Missing required fields' });
    }
    
    // Test connection first
    const isValid = await testConnection({
      instanceUrl: instance_url,
      username,
      password,
      securityToken: security_token,
    });
    
    if (!isValid) {
      return res.status(400).json({ error: 'Invalid credentials or connection failed' });
    }
    
    // Encrypt sensitive data
    const encryptedPassword = encrypt(password);
    const encryptedToken = security_token ? encrypt(security_token) : null;
    
    const result = await pool.query(
      `INSERT INTO connections (name, type, instance_url, username, encrypted_password, encrypted_security_token)
       VALUES ($1, $2, $3, $4, $5, $6)
       RETURNING id, name, type, instance_url, created_at`,
      [name, type, instance_url, username, encryptedPassword, encryptedToken]
    );
    
    res.status(201).json(result.rows[0]);
  } catch (error: any) {
    console.error('Error creating connection:', error);
    res.status(500).json({ error: 'Failed to create connection' });
  }
});

// GET /api/v1/connections - List all
router.get('/', async (req, res) => {
  try {
    const result = await pool.query(
      'SELECT id, name, type, instance_url, username, created_at FROM connections ORDER BY created_at DESC'
    );
    res.json(result.rows);
  } catch (error: any) {
    res.status(500).json({ error: 'Failed to fetch connections' });
  }
});

// POST /api/v1/connections/:id/test - Test connection
router.post('/:id/test', async (req, res) => {
  try {
    const { id } = req.params;
    const result = await pool.query(
      'SELECT instance_url, username, encrypted_password, encrypted_security_token FROM connections WHERE id = $1',
      [id]
    );
    
    if (result.rows.length === 0) {
      return res.status(404).json({ error: 'Connection not found' });
    }
    
    // Decrypt and test (decryption logic omitted for brevity)

    res.json({ success: isValid, message: isValid ? 'Connection successful' : 'Connection failed' });
  } catch (error: any) {
    res.status(500).json({ error: 'Failed to test connection' });
  }
});

export default router;
